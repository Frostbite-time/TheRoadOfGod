# 夺取对方的灵魂
super_take_his_soul = {
	icon = icon_hostile
	desc = super_take_his_soul_desc
	common_interaction = yes
	interface_priority = 90
	category = interaction_category_hostile
	auto_accept = yes
	ai_frequency = 6
	ai_targets = {
		ai_recipients = neighboring_rulers
	}
	ai_target_quick_trigger = {
		adult = yes
	}
	ai_will_do = {
		base = 0
		modifier = {
			add = 100
			scope:actor = {
				is_at_war_with = scope:recipient
			}
		}
	}
	ai_potential = {
		always = yes
	}
	ignores_pending_interaction_block = yes

	cooldown = {
		years = 3
	}

	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = {
			is_equie_soul_stone = yes
		}
	}

	is_valid_showing_failures_only = {
		NOT = {
			scope:recipient = {
				is_character_powerful = yes
			}
		}		
	}

	

	on_accept = {
		scope:recipient = {
			death = {
				death_reason = death_for_lost_your_soul
				killer = scope:actor
			}
		}
	}
}

# 使用毁灭之火
super_use_black_fire_interaction = {
	icon = icon_hostile
	desc = super_use_black_fire_interaction_desc
	common_interaction = yes
	interface_priority = 90
	category = interaction_category_hostile
	auto_accept = yes
	ai_frequency = 6
	ai_targets = {
		ai_recipients = neighboring_rulers
	}
	ai_target_quick_trigger = {
		adult = yes
	}
	ai_will_do = {
		base = 0
		modifier = {
			add = 100
			scope:actor = {
				is_at_war_with = scope:recipient
			}
		}
	}
	ai_potential = {
		always = yes
	}
	ignores_pending_interaction_block = yes

	cooldown = {
		days = 1
	}

	is_shown = {
		NOT = { scope:recipient = scope:actor }
		scope:actor = {
			has_black_fire = yes
		}
	}

	is_valid_showing_failures_only = {
		always = yes
	}

	

	on_accept = {
		scope:actor = {
			random_character_artifact = {
				limit = {
					has_artifact_modifier = dark_fire_modifier
				}
				destroy_artifact = this
			}
		}
		scope:recipient = {
			this.location = {
				duchy = {
					save_scope_as = 被焚毁的公国
				}
			}
			super_black_fire_destory_duchy = yes
		}

		every_living_character = {
			limit = {
				is_landed = yes
				NOT = {
					this = scope:recipient
				}
				NOT = {
					this = scope:actor
				}
			}
			trigger_event = {
				id = super_unknow_fire_events.0009
				days = { 1 20 }
			}

		}
	}
}
