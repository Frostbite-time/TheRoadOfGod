}

super_random_events.0003 = {
	type = character_event
	title = super_random_events.0003.t
	desc = super_random_events.0003.desc
	theme = realm

	override_background = {
        reference = super_random_events_3
    }
	option = {
		name = super_random_events.0003.1
		trigger = {
			has_trait = devilishness
		}
		hidden_effect = {
			every_close_family_member = {
				limit = {
					NOT = {
						是否拥有足够强大的实力 = yes
					}
				}
				death = {
					death_reason = death_for_murder_by_devilishness_close_one
					killer = root
				}
			}
			trigger_event = {
				id = super_random_events.0004
				days = { 1 6 }
			}
		}	
	}

	option = {
		name = super_random_events.0003.2
		trigger = {
			has_trait = saint_in_flesh_traits
		}
		hidden_effect = {
			trigger_event = super_random_events.0005
		}
	}
	

	option = {
		name = super_random_events.0003.3
		trigger = {
			has_trait = angel_traits
		}
		hidden_effect = {
			trigger_event = super_random_events.0006
		}
		
	}

	option = {
		name = super_random_events.0003.4
		hidden_effect = {
			death = {
				death_reason = death_for_lost_your_soul
			}
		}
	}
	
	option = {
		name = super_random_events.0003.5
		hidden_effect = {
			random_list = {
				1 ={
					death = {
						death_reason = death_for_lost_your_soul
					}
				}
				1 = {
					trigger_event = super_random_events.0007
				}
			}
		}
		
	}

	option = {
		name = super_random_events.0003.6
		hidden_effect = {
			trigger_event = super_random_events.0008
		}
	}
	
}
