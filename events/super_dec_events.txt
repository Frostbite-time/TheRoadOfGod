	left_portrait = root		#让角色出现在左边
	right_portrait = scope:god_ll_a
	immediate = {   #当事件触发时会发生的事情，优先级高于一切
		create_character = {
			name = god_of_destiny_name
			trait = super_destiny_god
			after_creation = {
				save_scope_as = god_ll_a   
			}
		}
	}
	option = {
		name = call_god_events.0003.a  #钱 
		root = {
			add_gold = 2000
		}
	}
	option = {
		name = call_god_events.0003.b		#威望
		root = {
			add_prestige = 2000
		}
	}
	option = {
		name = call_god_events.0003.c  #虔诚
		root = {
			add_piety = 2000
		}
	}
	option = {
		name = call_god_events.0003.d		#领地
		trigger_event = {
			id = call_god_events.0006
			days = { 3 6 }
		}
	}
	option = {
		name = call_god_events.0003.e  #压力
		add_stress = -300
	}
	after = {
		hidden_effect = {
			scope:god_ll_a = {
				death = {
					death_reason = death_vanished
				}
			}
		}
	}	
}
